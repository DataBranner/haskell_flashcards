\fl{factorial (recursive)}{
factorial &:: & Int \rightarrow Int\\
factorial~0 &= & 1\\
factorial~(n+1) &= & (n+1) * factorial~n
}


\fl{factorial (non-recursive)}{
factorial &:: & Int \rightarrow Int\\
factorial~n &= & product~[1..n]}

\fl{insertion sort}{
isort			&::	&	Ord~a~\Rightarrow~[a] \rightarrow [a]\\
isort~[]		&=	&	[]\\
isort~(x:xs)	&=	&	insert~x~(isort~xs)
}
